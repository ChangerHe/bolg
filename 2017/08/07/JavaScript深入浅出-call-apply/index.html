引用类型常量的赋值问题

当函数内部制定了变量值后,输出的值是不一样的,我们来看一下例子:
function foo(x) {
x.push(4);
console.log(x); //输出的是[1,2,3,4],没毛病
x = [4,5,6];
x.push(7);
console.log(x); //输出[4,5,6,7]这个没毛病
}
var a = [1,2,3];
foo(a);
console.log(a); //这个情况下,a的值其实是[1,2,3,4];

function foo(x) {
x = x + 1;
console.log(x);
}
var a = 2;
foo(a);
console.log(a); //此时a还是为2
```
